const MetronomeContract = require('../contracts/interface/metronome.sol.js')
const ContractReceiverContract = require('../contracts/interface/contractreceiver.sol.js')
const TokenContract = require('../contracts/interface/token.sol.js')
const ReserveTokenContract = require('../contracts/interface/reservetoken.sol.js')
const ERC223Contract = require('../contracts/interface/erc223.sol.js')
// const PairContract = require('../contracts/interface/pair.sol.js')
 
const buildContracts = context => {
  MetronomeContract.setProvider(context.provider)
  MetronomeContract.setProvider(context.provider)
  TokenContract.setProvider(context.provider)
  ReserveTokenContract.setProvider(context.provider)
  ERC223Contract.setProvider(context.provider)
  // PairContract.setProvider(context.provider)
 
  const instances = []
  instances.push(MetronomeContract.at(context.config.addr))
  instances.push(ContractReceiverContract.at(context.config.addr))
  instances.push(TokenContract.at(context.config.addr))
  instances.push(ReserveTokenContract.at(context.config.addr))
  instances.push(ERC223Contract.at(context.config.addr))
  // PairContract.at(context.config.addr)
 
  return Promise.all(
    instances
  ).then(([metronome, contractReceiver, token, reserveToken, erc223]) => ({
    metronome, contractReceiver, token, reserveToken, erc223
  })) // TODO: tmp
}
 
module.exports = buildContracts
